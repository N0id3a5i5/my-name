// ── Native HTTP helper (bypasses WebView CORS on Android)
// Uses CapacitorHttp natively; falls back to window.fetch in a browser.
const capFetch = async (url: string, options: {
  method: 'GET' | 'POST';
  headers?: Record<string, string>;
  body?: string;
}): Promise<{ ok: boolean; json: () => Promise<any> }> => {
  const isNative = !!(window as any).Capacitor?.isNativePlatform?.();
  if (isNative) {
    const res = await CapacitorHttp.request({
      url,
      method: options.method,
      headers: options.headers || {},
      data: options.body ? JSON.parse(options.body) : undefined,
      responseType: 'json',
    });
    return {
      ok: res.status >= 200 && res.status < 300,
      json: async () => (typeof res.data === 'string' ? JSON.parse(res.data) : res.data),
    };
  }
  const res = await fetch(url, {
    method: options.method,
    headers: options.headers,
    body: options.body,
  });
  return { ok: res.ok, json: () => res.json() };
};

// ── SQLite Storage Layer
// Uses @capacitor-community/sqlite natively on Android.
// Falls back to localStorage when running in a browser (dev / web build).
// ─────────────────────────────────────────────────────────────────────────────
const _isNative = (): boolean => !!(window as any).Capacitor?.isNativePlatform?.();
const _sqlite = new SQLiteConnection(CapacitorSQLite);
let _db: any = null;

const initDB = async (): Promise<void> => {
  if (!_isNative()) return; // web build — use localStorage fallback
  if (_db) return;          // already initialised
  try {
    const consistency = await _sqlite.checkConnectionsConsistency();
    const isConn = (await _sqlite.isConnection('mcq_db', false)).result;
    if (consistency.result && isConn) {
      _db = await _sqlite.retrieveConnection('mcq_db', false);
    } else {
      _db = await _sqlite.createConnection('mcq_db', false, 'no-encryption', 1, false);
    }
    await _db.open();
    await _db.execute(`
      CREATE TABLE IF NOT EXISTS settings (
        key   TEXT PRIMARY KEY,
        value TEXT NOT NULL DEFAULT ''
      );
      CREATE TABLE IF NOT EXISTS saved_tests (
        id        TEXT PRIMARY KEY,
        name      TEXT NOT NULL,
        saved_at  INTEGER NOT NULL,
        questions TEXT NOT NULL
      );
      CREATE TABLE IF NOT EXISTS attempt_history (
        id        TEXT PRIMARY KEY,
        test_id   TEXT NOT NULL,
        test_name TEXT NOT NULL,
        date      INTEGER NOT NULL,
        pct       INTEGER NOT NULL,
        correct   INTEGER NOT NULL,
        wrong     INTEGER NOT NULL,
        skipped   INTEGER NOT NULL,
        total     INTEGER NOT NULL,
        time_taken INTEGER NOT NULL
      );
    `);
  } catch (e) {
    console.error('SQLite init failed — falling back to localStorage', e);
    _db = null;
  }
};

// ── Settings (API keys etc.)
const dbGetSetting = async (key: string): Promise<string> => {
  if (!_isNative() || !_db) return localStorage.getItem(key) || '';
  try {
    const res = await _db.query('SELECT value FROM settings WHERE key = ?', [key]);
    return res.values?.[0]?.value || '';
  } catch { return ''; }
};
const dbSetSetting = async (key: string, value: string): Promise<void> => {
  if (!_isNative() || !_db) { try { localStorage.setItem(key, value); } catch {} return; }
  try { await _db.run('INSERT OR REPLACE INTO settings (key, value) VALUES (?, ?)', [key, value]); } catch {}
};

// ── Saved Tests
const dbGetSavedTests = async (): Promise<SavedTest[]> => {
  if (!_isNative() || !_db) {
    try { const r = localStorage.getItem('mcq_saved_tests'); return r ? JSON.parse(r) : []; } catch { return []; }
  }
  try {
    const res = await _db.query('SELECT * FROM saved_tests ORDER BY saved_at DESC');
    return (res.values || []).map((r: any) => ({
      id: r.id, name: r.name, savedAt: r.saved_at,
      questions: JSON.parse(r.questions),
    }));
  } catch { return []; }
};
const dbSaveTest = async (name: string, questions: Question[]): Promise<SavedTest> => {
  const entry: SavedTest = { id: Date.now().toString(), name, savedAt: Date.now(), questions };
  if (!_isNative() || !_db) {
    try {
      const tests = await dbGetSavedTests();
      const idx = tests.findIndex(t => t.name === name);
      if (idx >= 0) { entry.id = tests[idx].id; tests[idx] = entry; } else tests.unshift(entry);
      localStorage.setItem('mcq_saved_tests', JSON.stringify(tests));
    } catch {}
    return entry;
  }
  try {
    const existing = await _db.query('SELECT id FROM saved_tests WHERE name = ?', [name]);
    if (existing.values?.length) {
      entry.id = existing.values[0].id;
      await _db.run('UPDATE saved_tests SET saved_at = ?, questions = ? WHERE id = ?',
        [entry.savedAt, JSON.stringify(questions), entry.id]);
    } else {
      await _db.run('INSERT INTO saved_tests (id, name, saved_at, questions) VALUES (?, ?, ?, ?)',
        [entry.id, name, entry.savedAt, JSON.stringify(questions)]);
    }
  } catch {}
  return entry;
};
const dbDeleteTest = async (id: string): Promise<void> => {
  if (!_isNative() || !_db) {
    try {
      const tests = (await dbGetSavedTests()).filter(t => t.id !== id);
      localStorage.setItem('mcq_saved_tests', JSON.stringify(tests));
    } catch {}
    return;
  }
  try { await _db.run('DELETE FROM saved_tests WHERE id = ?', [id]); } catch {}
};

// ── Attempt History
const dbGetAttempts = async (testId?: string): Promise<AttemptRecord[]> => {
  if (!_isNative() || !_db) {
    try {
      const all: AttemptRecord[] = JSON.parse(localStorage.getItem('mcq_attempt_history') || '[]');
      return testId ? all.filter(a => a.testId === testId) : all;
    } catch { return []; }
  }
  try {
    const res = testId
      ? await _db.query('SELECT * FROM attempt_history WHERE test_id = ? ORDER BY date DESC', [testId])
      : await _db.query('SELECT * FROM attempt_history ORDER BY date DESC LIMIT 200');
    return (res.values || []).map((r: any) => ({
      id: r.id, testId: r.test_id, testName: r.test_name,
      date: r.date, pct: r.pct, correct: r.correct, wrong: r.wrong,
      skipped: r.skipped, total: r.total, timeTaken: r.time_taken,
    }));
  } catch { return []; }
};
const dbSaveAttempt = async (record: Omit<AttemptRecord, 'id'>): Promise<AttemptRecord> => {
  const entry = { ...record, id: Date.now().toString() };
  if (!_isNative() || !_db) {
    try {
      const all = await dbGetAttempts();
      all.unshift(entry);
      localStorage.setItem('mcq_attempt_history', JSON.stringify(all.slice(0, 100)));
    } catch {}
    return entry;
  }
  try {
    await _db.run(
      `INSERT INTO attempt_history
        (id, test_id, test_name, date, pct, correct, wrong, skipped, total, time_taken)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      [entry.id, entry.testId, entry.testName, entry.date, entry.pct,
       entry.correct, entry.wrong, entry.skipped, entry.total, entry.timeTaken]
    );
  } catch {}
  return entry;
};
const dbDeleteAttempts = async (testId: string): Promise<void> => {
  if (!_isNative() || !_db) {
    try {
      const remaining = (await dbGetAttempts()).filter(a => a.testId !== testId);
      localStorage.setItem('mcq_attempt_history', JSON.stringify(remaining));
    } catch {}
    return;
  }
  try { await _db.run('DELETE FROM attempt_history WHERE test_id = ?', [testId]); } catch {}
};

// ── pdf.js helpers (bundled — no CDN, works in Capacitor/Android)
// Detects scanned pages (image-only) and renders them to base64 PNG for OCR.
// Returns per-page text array + scanned page images so callers can use vision on blank pages.
//
// Fix #1  — Scanned pages: render to canvas → base64, stored in scannedPageImages[pageIdx].
// Fix #2  — Hindi/Devanagari: getTextContent with normalizeWhitespace:true preserves ligatures.
// Fix #4  — Two-column layout (VisionIAS, UPSC papers): detect column boundary and reconstruct
//           left column first, then right column, so Q1-Q50 left + Q51-Q100 right reads correctly.
// ── v14: Detect garbled text from legacy Devanagari fonts (Kruti Dev, Chanakya, Shivaji01).
// These fonts store Roman chars internally but render as Devanagari visually.
// pdf.js extracts "dk;Z" instead of "कार्य" — high semicolon-to-letter ratio is a strong signal.
const isGarbledText = (text: string): boolean => {
  if (text.length < 20) return false;
  const asciiLetters = (text.match(/[a-zA-Z]/g) || []).length;
  const semicolons = (text.match(/;/g) || []).length;
  // Kruti Dev signature: lots of semicolons mixed with ASCII
  if (asciiLetters > 20 && semicolons / asciiLetters > 0.08) return true;
  // Private Use Area / replacement chars indicate broken font mapping
  const garbledChars = (text.match(/[\uFFFD\uE000-\uF8FF]/g) || []).length;
  if (garbledChars / text.length > 0.05) return true;
  // High density of backslash / caret / tilde — another Kruti Dev artifact
  const oddChars = (text.match(/[\\^~|#@]/g) || []).length;
  if (asciiLetters > 30 && oddChars / asciiLetters > 0.12) return true;
  // Almost no spaces relative to length — likely symbol-font scramble
  const spaces = (text.match(/ /g) || []).length;
  if (text.length > 60 && spaces / text.length < 0.03) return true;
  return false;
};

const extractPdfPages = async (
  file: File
): Promise<{ pages: string[]; scannedPageCount: number; scannedPageImages: Record<number, string> }> => {
  const buf = await file.arrayBuffer();
  const pdf = await pdfjsLib.getDocument({ data: buf, verbosity: 0 }).promise
    .catch((err: any) => {
      if (err?.name === 'PasswordException' || err?.message?.includes('password')) {
        throw new Error('This PDF is password-protected. Please unlock it first and try again.');
      }
      throw err;
    });
  const pages: string[] = [];
  let scannedPageCount = 0;
  const scannedPageImages: Record<number, string> = {};

  // Render scanned page to base64 — sequential to avoid OOM on large PDFs.
  // Scale 1.5 instead of 2.0: still sufficient for OCR, uses ~44% less memory.
  const renderPageToBase64 = async (page: any): Promise<string> => {
    const vp = page.getViewport({ scale: 1.5 });
    const canvas = document.createElement('canvas');
    canvas.width = vp.width; canvas.height = vp.height;
    await page.render({ canvasContext: canvas.getContext('2d')!, viewport: vp }).promise;
    const b64 = canvas.toDataURL('image/jpeg', 0.85).split(',')[1]; // JPEG saves ~3× vs PNG
    canvas.width = 0; canvas.height = 0; // free GPU memory immediately
    return b64;
  };

  for (let i = 1; i <= pdf.numPages; i++) {
    const page = await pdf.getPage(i);
    const viewport = page.getViewport({ scale: 1.0 });
    const pageWidth = viewport.width;
    const content = await page.getTextContent({ normalizeWhitespace: true } as any);
    const items = content.items as any[];

    if (items.length === 0) {
      pages.push('');
      scannedPageCount++;
      try { scannedPageImages[i - 1] = await renderPageToBase64(page); } catch {}
      continue;
    }

    // ── Column detection via gap analysis instead of hard midpoint.
    // Find the largest horizontal gap between text items — that gap IS the column gutter.
    // This works for asymmetric layouts (40/60 split etc.) unlike the old pageWidth/2 approach.
    const xs = items.map((it: any) => it.transform[4]).sort((a, b) => a - b);
    const pageRight = pageWidth * 0.9;
    const pageLeft  = pageWidth * 0.1;
    // Find largest gap between consecutive x positions in the middle 80% of the page
    let colBoundary = pageWidth / 2; // fallback
    let maxGap = 0;
    for (let j = 1; j < xs.length; j++) {
      const gap = xs[j] - xs[j - 1];
      const midPoint = (xs[j] + xs[j - 1]) / 2;
      if (gap > maxGap && midPoint > pageLeft && midPoint < pageRight) {
        maxGap = gap;
        colBoundary = midPoint;
      }
    }
    // Only treat as multi-column if gap is significant (>8% of page width)
    const isTwoColumn = maxGap > pageWidth * 0.08
      && items.filter((it: any) => it.transform[4] < colBoundary).length > 5
      && items.filter((it: any) => it.transform[4] >= colBoundary).length > 5;

    // Three-column: find two gaps
    let col2Boundary = pageWidth * 0.67;
    let isThreeColumn = false;
    if (isTwoColumn) {
      const leftItems_  = items.filter((it: any) => it.transform[4] < colBoundary);
      const rightItems_ = items.filter((it: any) => it.transform[4] >= colBoundary);
      const rxs = rightItems_.map((it: any) => it.transform[4]).sort((a, b) => a - b);
      let maxGap2 = 0;
      for (let j = 1; j < rxs.length; j++) {
        const gap = rxs[j] - rxs[j - 1];
        const mid = (rxs[j] + rxs[j - 1]) / 2;
        if (gap > maxGap2 && mid > colBoundary && mid < pageRight) { maxGap2 = gap; col2Boundary = mid; }
      }
      isThreeColumn = maxGap2 > pageWidth * 0.08
        && rightItems_.filter((it: any) => it.transform[4] < col2Boundary).length > 5
        && rightItems_.filter((it: any) => it.transform[4] >= col2Boundary).length > 5;
    }

    const reconstruct = (itemSet: any[]): string => {
      const sorted = [...itemSet].sort((a: any, b: any) => {
        const dy = Math.round(b.transform[5] / 5) - Math.round(a.transform[5] / 5);
        if (dy !== 0) return dy;
        return a.transform[4] - b.transform[4];
      });
      let text = '';
      let lastY: number | null = null;
      for (const item of sorted) {
        const y = Math.round(item.transform[5]);
        if (lastY !== null && Math.abs(y - lastY) > 10) text += '\n';
        text += (item.str || '') + ' ';
        lastY = y;
      }
      return text.trim();
    };

    let text: string;
    if (isThreeColumn) {
      const c1 = reconstruct(items.filter((it: any) => it.transform[4] < colBoundary));
      const c2 = reconstruct(items.filter((it: any) => it.transform[4] >= colBoundary && it.transform[4] < col2Boundary));
      const c3 = reconstruct(items.filter((it: any) => it.transform[4] >= col2Boundary));
      text = [c1, c2, c3].filter(Boolean).join('\n');
    } else if (isTwoColumn) {
      const leftText  = reconstruct(items.filter((it: any) => it.transform[4] < colBoundary));
      const rightText = reconstruct(items.filter((it: any) => it.transform[4] >= colBoundary));
      text = leftText + (leftText && rightText ? '\n' : '') + rightText;
    } else {
      text = reconstruct(items);
    }

    pages.push(text);

    // Scanned/garbled: threshold raised to 50 chars (avoids false-positives on header-only pages)
    if (text.length < 50 || isGarbledText(text)) {
      scannedPageCount++;
      try { scannedPageImages[i - 1] = await renderPageToBase64(page); } catch {}
    }
  }
  return { pages, scannedPageCount, scannedPageImages };
};

// ── F3: Strip repeated headers, footers, and watermark text from pages.
// Detects lines that appear on >50% of pages (headers/footers) or >40% (watermarks).
const stripHeadersFooters = (pages: string[]): string[] => {
  if (pages.length < 3) return pages;
  const firstLines = pages.map(p => p.split('\n')[0]?.trim() || '');
  const lastLines  = pages.map(p => { const l = p.split('\n'); return l[l.length - 1]?.trim() || ''; });
  const countRepeats = (lines: string[]) => {
    const freq: Record<string, number> = {};
    lines.forEach(l => { if (l.length > 3 && l.length < 200) freq[l] = (freq[l] || 0) + 1; });
    return Object.entries(freq).filter(([, c]) => c > pages.length * 0.5).map(([l]) => l);
  };
  const headerPatterns = countRepeats(firstLines);
  const footerPatterns = countRepeats(lastLines);
  // Watermark-like lines: appear on >40% of pages but not as first/last line
  const allLines = pages.flatMap(p => p.split('\n').map(l => l.trim()));
  const lineFreq: Record<string, number> = {};
  allLines.forEach(l => { if (l.length > 5 && l.length < 100) lineFreq[l] = (lineFreq[l] || 0) + 1; });
  const watermarks = Object.entries(lineFreq)
    .filter(([, c]) => c > pages.length * 0.4)
    .filter(([l]) => !headerPatterns.includes(l) && !footerPatterns.includes(l))
    .map(([l]) => l);
  const removeSet = new Set([...headerPatterns, ...footerPatterns, ...watermarks]);
  if (removeSet.size > 0) {
    return pages.map(p => p.split('\n').filter(l => !removeSet.has(l.trim())).join('\n'));
  }
  return pages;
};

// ── F5: Extract answer key from last pages before slicing.
// Returns answers map AND how many tail pages were consumed (so callers can exclude them from AI slices).
const extractAnswerKey = (pages: string[]): { answers: Record<number, string>; tailPageCount: number } | null => {
  const answers: Record<number, string> = {};
  // Scan up to last 5 pages — some papers have 2–3 page answer keys
  const tailCount = Math.min(5, pages.length);
  const tail = pages.slice(-tailCount).join('\n');
  const regex = /(?:Q\.?\s*|Ans[:.\s]?\s*)?(?:^|\s)(\d{1,3})\s*[.):\-–]\s*\(?([A-Da-d])\)?\b/gm;
  let match;
  while ((match = regex.exec(tail)) !== null) {
    const qNum = parseInt(match[1]);
    if (qNum > 0 && qNum <= 300) answers[qNum] = match[2].toUpperCase();
  }
  if (Object.keys(answers).length < 5) return null;
  // Determine how many of the tail pages actually contain answer key lines
  let pagesWithAnswers = 0;
  for (let i = pages.length - 1; i >= pages.length - tailCount; i--) {
    const pgAnswers: number[] = [];
    let m2;
    const re2 = /(?:Q\.?\s*)?(?:^|\s)(\d{1,3})\s*[.):\-–]\s*\(?([A-Da-d])\)?\b/gm;
    while ((m2 = re2.exec(pages[i])) !== null) pgAnswers.push(parseInt(m2[1]));
    if (pgAnswers.length >= 3) pagesWithAnswers++;
    else break; // stop as soon as a non-answer page is hit
  }
  return { answers, tailPageCount: Math.max(1, pagesWithAnswers) };
};

// ── Stitch page text so questions are never silently split at slice boundaries.
// Slices break at question-number boundaries (Q47., 47., etc.) when possible,
// so a question is never split across two slices. Falls back to newline, then hard cut.
const buildSlicesWithOverlap = (pages: string[], targetCharsPerSlice = 12000, overlapChars = 1200): string[] => {
  const fullText = pages.join('\n');
  const slices: string[] = [];
  let currentStart = 0;

  // Regex: a question number at the start of a line, e.g. "47.", "Q47.", "Q. 47", "47)"
  const qBoundary = /(?:^|\n)\s*(?:Q\.?\s*)?\d{1,3}\s*[.)]/g;

  while (currentStart < fullText.length) {
    let currentEnd = Math.min(currentStart + targetCharsPerSlice, fullText.length);

    if (currentEnd < fullText.length) {
      // 1st preference: break just before the NEXT question number within ±600 chars of target
      const searchWindow = fullText.slice(currentEnd - 600, currentEnd + 600);
      qBoundary.lastIndex = 0;
      let bestQBreak = -1;
      let m: RegExpExecArray | null;
      while ((m = qBoundary.exec(searchWindow)) !== null) {
        const abs = currentEnd - 600 + m.index;
        if (abs > currentStart + targetCharsPerSlice * 0.5) bestQBreak = abs; // take the last one before target
      }
      if (bestQBreak > currentStart) {
        currentEnd = bestQBreak;
      } else {
        // 2nd preference: nearest newline within +500 chars
        const nextNewline = fullText.indexOf('\n', currentEnd);
        if (nextNewline !== -1 && nextNewline - currentEnd < 500) currentEnd = nextNewline;
      }
    }

    let sliceText = fullText.substring(currentStart, currentEnd);
    if (currentStart > 0) {
      const prevStart = Math.max(0, currentStart - overlapChars);
      const overlapText = fullText.substring(prevStart, currentStart);
      sliceText = `[...continued from previous section]\n${overlapText}\n` + sliceText;
    }

    if (sliceText.trim().length > 20) slices.push(sliceText.trim());
    currentStart = currentEnd;
  }
  return slices;
};

// Convenience: full text (used by single-provider paths)
const extractPdfText = async (file: File): Promise<string> => {
  const { pages } = await extractPdfPages(file);
  return pages.join('\n').trim();
};

// ── Fix #1: OCR individual scanned pages via Gemini vision.
// Called when extractPdfPages finds image-only pages and we have a Gemini key.
// Returns OCR'd text for each scanned page index, merged back into the pages array.
const ocrScannedPagesWithGemini = async (
  scannedPageImages: Record<number, string>,
  apiKey: string,
  addLog: (msg: string, kind: 'ok' | 'err' | 'warn' | 'info') => void
): Promise<Record<number, string>> => {
  const models = ['models/gemini-2.5-flash'];
  const pageIndices = Object.keys(scannedPageImages).map(Number);
  addLog(`[OCR] Running Gemini vision OCR on ${pageIndices.length} scanned page(s)...`, 'info');
  const recovered: Record<number, string> = {};

  for (const pageIdx of pageIndices) {
    const base64 = scannedPageImages[pageIdx];
    if (!base64) continue;
    let attempts = 0;
    while (attempts < 5 && !recovered[pageIdx]) {
      for (const model of models) {
        try {
          const res = await capFetch(
            `https://generativelanguage.googleapis.com/v1beta/${model}:generateContent?key=${apiKey}`,
            {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify({
                contents: [{
                  parts: [
                    { text: 'Extract ALL text from this exam page exactly as it appears. Preserve question numbers, options (A/B/C/D or 1/2/3/4), and Hindi/English text. Return plain text only.' },
                    { inline_data: { mime_type: 'image/png', data: base64 } },
                  ],
                }],
                generationConfig: { temperature: 0.1 },
              }),
            }
          );
          const data = await res.json();
          if (data.error) {
            if (data.error.code === 429 || data.error.message?.toLowerCase().includes('quota') || data.error.message?.toLowerCase().includes('rate limit')) {
              attempts++;
              const waitMs = 15000 * attempts; // Wait longer for Gemini quota limits
              addLog(`[OCR] Gemini Page ${pageIdx + 1} rate-limited (attempt ${attempts}/5). Waiting ${waitMs / 1000}s...`, 'warn');
              await new Promise(r => setTimeout(r, waitMs));
              break; // Break inner loop to retry outer loop
            }
            continue; // Move to next model
          }
          if (!data.candidates?.[0]?.content?.parts?.[0]?.text) continue;
          recovered[pageIdx] = data.candidates[0].content.parts[0].text;
          addLog(`[OCR] Page ${pageIdx + 1} recovered (${recovered[pageIdx].length} chars)`, 'ok');
          break; // Success, break models loop
        } catch (e: any) {
          if (e.message?.toLowerCase().includes('rate') || e.message?.toLowerCase().includes('too many') || e.message?.includes('429')) {
            attempts++;
            const waitMs = 15000 * attempts;
            addLog(`[OCR] Gemini Page ${pageIdx + 1} rate-limited (attempt ${attempts}/5). Waiting ${waitMs / 1000}s...`, 'warn');
            await new Promise(r => setTimeout(r, waitMs));
            break; // Break inner loop to retry outer loop
          }
          continue;
        }
      }
      // If we recovered the text or exhausted models without a rate limit, we can stop retrying
      if (recovered[pageIdx] || attempts === 0) break;
    }
    if (!recovered[pageIdx]) addLog(`[OCR] Page ${pageIdx + 1} OCR failed — skipped`, 'warn');
  }
  return recovered;
};

// ── v13: OCR scanned pages via NVIDIA vision (llama-3.2-11b-vision-instruct).
// Preferred over Gemini OCR when nvidia key is available — preserves Gemini quota.
// Uses OpenAI-compatible chat/completions with image_url content type.
const ocrScannedPagesWithNvidia = async (
  scannedPageImages: Record<number, string>,
  nvKey: string,
  addLog: (msg: string, kind: 'ok' | 'err' | 'warn' | 'info') => void
): Promise<Record<number, string>> => {
  const model = 'meta/llama-3.2-11b-vision-instruct';
  const pageIndices = Object.keys(scannedPageImages).map(Number);
  addLog(`[OCR] Running NVIDIA vision OCR on ${pageIndices.length} scanned page(s)...`, 'info');
  const recovered: Record<number, string> = {};

  for (const pageIdx of pageIndices) {
    const base64 = scannedPageImages[pageIdx];
    if (!base64) continue;
    let attempts = 0;
    while (attempts < 5) {
      try {
        const res = await capFetch('https://integrate.api.nvidia.com/v1/chat/completions', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${nvKey}` },
          body: JSON.stringify({
            model,
            messages: [{
              role: 'user',
              content: [
                { type: 'text', text: 'Extract ALL text from this exam page exactly as it appears. Preserve question numbers, options (A/B/C/D or 1/2/3/4), and Hindi/English text. Return plain text only.' },
                { type: 'image_url', image_url: { url: `data:image/png;base64,${base64}` } },
              ],
            }],
            max_tokens: 2048,
            temperature: 0.1,
          }),
        });
        const data = await res.json();

        if (data.error) {
          if (data.error.code === 429 || data.error.message?.toLowerCase().includes('rate')) {
            attempts++;
            const waitMs = 15000 * attempts; // Wait longer for strict API quotas
            addLog(`[OCR] NVIDIA Page ${pageIdx + 1} rate-limited (attempt ${attempts}/5). Waiting ${waitMs / 1000}s...`, 'warn');
            await new Promise(r => setTimeout(r, waitMs));
            continue;
          }
          throw new Error(data.error.message);
        }

        const text = data.choices?.[0]?.message?.content || '';
        if (!text) { addLog(`[OCR] NVIDIA Page ${pageIdx + 1} empty — skipped`, 'warn'); break; }
        recovered[pageIdx] = text;
        addLog(`[OCR] NVIDIA Page ${pageIdx + 1} recovered (${text.length} chars)`, 'ok');
        break; // Success, break retry loop
      } catch (e: any) {
        if (e.message?.toLowerCase().includes('rate') || e.message?.toLowerCase().includes('too many') || e.message?.includes('429')) {
          attempts++;
          const waitMs = 15000 * attempts;
          addLog(`[OCR] NVIDIA Page ${pageIdx + 1} rate-limited (attempt ${attempts}/5). Waiting ${waitMs / 1000}s...`, 'warn');
          await new Promise(r => setTimeout(r, waitMs));
          continue;
        }
        addLog(`[OCR] NVIDIA Page ${pageIdx + 1} error: ${e.message} — skipped`, 'warn');
        break; // Other error, break retry loop
      }
    }
  }
  return recovered;
};

// ── Quota / rate-limit detector
const isQuotaError = (data: any): boolean => {
  const msg: string = data?.error?.message || '';
  return (
    data?.error?.code === 429 ||
    msg.toLowerCase().includes('quota') ||
    msg.toLowerCase().includes('rate limit') ||
    msg.toLowerCase().includes('resource_exhausted')
  );
};

// ── FIX: Partial JSON recovery for truncated Groq responses
// When max_tokens is hit mid-object, we salvage all COMPLETE objects before the cut.
const recoverPartialJsonArray = (raw: string): Question[] => {
  const start = raw.indexOf('[');
  if (start === -1) return [];
  const content = raw.slice(start + 1);
  const questions: Question[] = [];
  let depth = 0;
  let objStart = -1;
  for (let i = 0; i < content.length; i++) {
    const ch = content[i];
    if (ch === '{') { if (depth === 0) objStart = i; depth++; }
    else if (ch === '}') {
      depth--;
      if (depth === 0 && objStart !== -1) {
        try {
          const obj = JSON.parse(content.slice(objStart, i + 1));
          // Keep questions even without answer (answer key backfill may fill it)
          if (obj.question && obj.options) questions.push(obj as Question);
        } catch {}
        objStart = -1;
      }
    }
  }
  return questions;
};

// ── Robust JSON extraction from LLM responses
const parseQuestionsFromLLM = (raw: string): Question[] => {
  let cleaned = raw
    .replace(/```json[\s\S]*?```/g, m => m.slice(7, -3))
    .replace(/```[\s\S]*?```/g, m => m.slice(3, -3))
    .trim()
    .replace(/`/g, '');

  // Strip any preamble text before the JSON array (e.g. "Here are the questions:\n[...")
  // Models sometimes emit disclaimers or affirmations before the actual JSON
  const bracketStart = cleaned.indexOf('[');
  if (bracketStart > 0) {
    const preamble = cleaned.slice(0, bracketStart);
    // Only strip if the preamble is clearly not part of JSON (no closing ] before [)
    if (!preamble.includes(']')) cleaned = cleaned.slice(bracketStart);
  }

  const sanitize = (s: string) => s
    .replace(/,(\s*[}\]])/g, '$1')
    .replace(/[\u201C\u201D]/g, '"').replace(/[\u2018\u2019]/g, "'")
    .replace(/[\u0000-\u0008\u000B\u000C\u000E-\u001F\u007F]/g, '')
    .replace(/\r/g, '');

  const start = cleaned.indexOf('[');
  const end = cleaned.lastIndexOf(']');

  if (start !== -1 && end !== -1 && end > start) {
    const slice = sanitize(cleaned.slice(start, end + 1));
    try {
      const parsed = JSON.parse(slice);
      if (Array.isArray(parsed)) return parsed as Question[];
      const arr = (Object.values(parsed) as any[]).find(v => Array.isArray(v));
      if (arr) return arr as Question[];
    } catch {
      try {
        const fixed = slice
          .replace(/(['"])?([a-zA-Z_][a-zA-Z0-9_]*)(['"])?:/g, '"$2":')
          .replace(/:\s*'([^']*)'/g, ': "$1"');
        const parsed = JSON.parse(fixed);
        if (Array.isArray(parsed)) return parsed as Question[];
      } catch {}
    }
  }

  // Partial recovery — handles token-limit truncation
  const recovered = recoverPartialJsonArray(cleaned);
  if (recovered.length > 0) return recovered;

  throw new Error('No JSON array found in model response');
};

// ── Normalize non-standard option labels to A/B/C/D
// Fix #5: Guard against statement-numbered questions (1. 2. 3. inside question body).
//         We only remap numeric/roman keys when they appear as OPTION KEYS, not when
//         the question body itself uses numbers for sub-statements.
// Fix #6: Roman numeral collision — i/ii/iii/iv are only treated as option labels
//         when ALL keys in the options object follow that pattern consistently.
const normalizeOptions = (q: any): { options: Record<string, string>; answer: string } => {
  const raw = q.options || {};
  const keys = Object.keys(raw);

  // Already standard A/B/C/D (case-insensitive check)
  // F9: Accept A-F as standard labels (not just A-D)
  const isStandard = keys.every(k => /^[A-Fa-f]$/.test(k.trim()));
  if (isStandard) {
    const norm: Record<string, string> = {};
    keys.forEach(k => { norm[k.trim().toUpperCase()] = raw[k]; });
    const ans = (q.answer || '').trim().toUpperCase();
    return { options: norm, answer: ans };
  }

  // Fix #6: Only apply roman numeral remapping if ALL keys are roman numerals.
  // This prevents i/ii in a mixed set from being silently remapped.
  const allRoman = keys.every(k => /^\(?(?:i{1,3}|iv|v?i{0,3})\)?$/i.test(k.trim()));
  const allNumeric = keys.every(k => /^\(?\d+\)?$/.test(k.trim()));

  // F9: Support up to 6 options (A-F) for GATE, CAT, etc.
  const LABEL_MAP: Record<string, string> = {
    '1': 'A', '2': 'B', '3': 'C', '4': 'D', '5': 'E', '6': 'F',
    '(1)': 'A', '(2)': 'B', '(3)': 'C', '(4)': 'D', '(5)': 'E', '(6)': 'F',
    'a': 'A', 'b': 'B', 'c': 'C', 'd': 'D', 'e': 'E', 'f': 'F',
    '(a)': 'A', '(b)': 'B', '(c)': 'C', '(d)': 'D', '(e)': 'E', '(f)': 'F',
    ...(allRoman ? {
      'i': 'A', 'ii': 'B', 'iii': 'C', 'iv': 'D', 'v': 'E', 'vi': 'F',
      '(i)': 'A', '(ii)': 'B', '(iii)': 'C', '(iv)': 'D', '(v)': 'E', '(vi)': 'F',
    } : {}),
  };

  // Fix #5: If numeric keys are used BUT the question body already has numbered
  // sub-statements (e.g. "1. India  2. Pakistan  3. …"), the numeric keys are
  // option labels for those statements and we remap them only if allNumeric holds.
  // If not allNumeric, fall back to positional A/B/C/D assignment.
  const normOptions: Record<string, string> = {};
  const keyRemap: Record<string, string> = {};

  keys.forEach((k, idx) => {
    const stripped = k.trim().toLowerCase();
    const mapped = LABEL_MAP[stripped] || (allNumeric || allRoman ? String.fromCharCode(65 + idx) : String.fromCharCode(65 + idx));
    normOptions[mapped] = raw[k];
    keyRemap[stripped] = mapped;
    keyRemap[k.trim()] = mapped;
    keyRemap[k.trim().toUpperCase()] = mapped;
  });

  // Remap answer key
  const rawAns = (q.answer || '').trim();
  const normAns =
    keyRemap[rawAns.toLowerCase()] ||
    keyRemap[rawAns] ||
    rawAns.toUpperCase();

  return { options: normOptions, answer: normAns };
};

// ── Detect and reformat Assertion-Reason / Match-the-Column style questions
// Fix #9: Expanded regex covers all UPSC/SSC variants:
//   "Both A and R are true", "A is correct but R is incorrect",
//   "R is the correct explanation", "Neither A nor R", etc.
const isAssertionReason = (q: any): boolean => {
  const opts = Object.values(q.options || {}) as string[];
  const combined = opts.join(' ') + ' ' + (q.question || '');
  return /both.*true|both.*correct|assertion.*reason|neither.*nor|a is correct|r is correct|r is the correct explanation|a is true.*r is false|a is false.*r is true/i.test(combined);
};

const validateQuestions = (parsed: Question[]): Question[] => {
  const valid: Question[] = [];
  for (const q of parsed) {
    if (!q.question || !q.options) continue;

    const { options, answer } = normalizeOptions(q);
    const optCount = Object.keys(options).length;

    if (optCount < 2) continue;
    if (answer && !options[answer]) continue;

    q.options = options;
    q.answer = answer || '';

    // Detect truncated Assertion-Reason options: option text ends mid-word or mid-sentence
    // (e.g. "Both A and R are true and R is the correct" — missing "explanation of A")
    // Flag these so the deduplication step can prefer the longer version if a duplicate exists
    const arOpts = Object.values(options);
    const hasTruncatedAR = isAssertionReason(q) && arOpts.some(v => {
      const trimmed = v.trim();
      return trimmed.length > 10 && !/[.?!)]$/.test(trimmed); // doesn't end on sentence boundary
    });
    if (hasTruncatedAR) (q as any)._truncatedAR = true;

    // Flag image-dependent questions
    if (/\[Visual question\]/i.test(q.question) || /(?:refer to the|given|see the)\s+(?:diagram|figure|graph|chart|map|image|picture)/i.test(q.question)) {
      (q as any)._hasVisual = true;
    }

    valid.push(q);
  }
  return valid;
};

    const slicePromises = groqSlices.map((sliceText, i) => {
      const { model, maxTokens } = GROQ_MODELS[i % GROQ_MODELS.length];
      return groqCallOne(sliceText, i, groqSlices.length, model, maxTokens, gKey)
        .catch(() => [] as Question[]);
    });
    const results = await Promise.all(slicePromises);
    const zeroSlices = results.filter(r => r.length === 0).length;
    const all = results.flat();
    addLog(`[Groq] Total from all slices: ${all.length} questions (${zeroSlices}/${groqSlices.length} slices returned 0)`, all.length > 0 ? 'ok' : 'warn');
    // All slices failed — almost certainly rate-limited across all models simultaneously
    if (all.length === 0 && groqSlices.length > 0) {
      addLog(`[Groq] ⚠ ALL slices returned 0 questions — likely rate-limited on all models. Try again in 60s or add a Gemini/NVIDIA key.`, 'err');
      throw new Error('Groq rate-limited on all models. Wait 60 seconds and retry, or add a Gemini or NVIDIA key as fallback.');
    }
    return all;
  };

  // ── GEMINI text-based extraction (used for split-work parallel mode)
  // Accepts raw text slice instead of a PDF file — faster, no base64 upload needed.
  const tryGeminiText = async (text: string, key: string, label: string): Promise<Question[] | null> => {
    // Only gemini-2.5-flash — removed 2.0-flash and 2.0-flash-lite per user request.
    const priority = ['gemini-2.5-flash'];
    const models = priority.map(p => `models/${p}`);

    // Gemini text chunks with 1500-char overlap to avoid splitting questions at boundaries
    const CHUNK_SIZE = 30000;
    const CHUNK_OVERLAP = 1500;
    const chunks: string[] = [];
    for (let i = 0; i < text.length; i += CHUNK_SIZE - CHUNK_OVERLAP) {
      chunks.push(text.slice(i, i + CHUNK_SIZE));
      if (i + CHUNK_SIZE >= text.length) break;
    }

    const allQuestions: Question[] = [];

    for (const model of models) {
      const shortName = model.split('/').pop() || model;
      addLog(`[${label}] Trying ${shortName}...`, 'info');

      let modelOk = true;
      for (let ci = 0; ci < chunks.length; ci++) {
        const chunk = chunks[ci];
        const prompt =
          `You are an MCQ extractor for Indian competitive exams (UPSC, SSC, GATE, NEET, JEE, State PSCs).\n` +
          `Extract ALL multiple-choice questions from the text below.\n` +
          `CRITICAL — THIS IS A UPSC/COMPETITIVE EXAM PAPER. COMMON PATTERNS:\n` +
          `- The paper likely has TWO COLUMNS per page. Left column has questions (e.g. Q1, Q3, Q5), right column has more (e.g. Q2, Q4, Q6). Read ALL questions from BOTH columns — do NOT miss the right column.\n` +
          `- The answer key appears at the END as "Q 1. C", "Q 2. B" etc. Match each question number to its answer letter.\n` +
          `- If this chunk contains an answer key section (lines like "Q 1. C  Q 2. B"), extract answers and pair with questions found in this or other chunks.\n` +
          `EXTRACTION RULES:\n` +
          `- Include ALL question types: standard MCQ, True/False, Assertion-Reason (Statement-I/Statement-II), Match-the-Column, Statement-based.\n` +
          `- If options are labelled 1/2/3/4 or a/b/c/d, normalize to A/B/C/D. Use roman numerals only if ALL options follow that scheme.\n` +
          `- For "Which of the following statements is/are correct?" questions, numbered sub-statements (1. 2. 3.) are PART of the question text, NOT option labels.\n` +
          `- The "answer" field MUST be one of the option letters (A, B, C, D, E, or F).\n` +
          `- DO NOT SKIP ANY QUESTION. Extract EVERY SINGLE MCQ, even if it seems incomplete or unusual.\n` +
          `- IMPORTANT: Do NOT output just a subset of questions. Extract EVERY SINGLE QUESTION you find. If the text has 50 questions, your JSON array MUST contain 50 objects.\n` +
          `- If a question references a diagram/figure/graph/map, include it and prefix with "[Visual question]".\n` +
          `- For Assertion-Reason: preserve option text exactly e.g. "Both Statement-I and Statement-II are correct and Statement-II is the correct explanation".\n` +
          `- For Match-the-Column: encode List-I vs List-II in the question text. A/B/C/D = matching combos.\n` +
          `- Respond with ONLY a JSON array. Do NOT include markdown, code fences, or any explanation.\n` +
          `- Each item: {"id":1,"question":"...","options":{"A":"...","B":"...","C":"...","D":"..."},"answer":"B"}\n` +
          `- If no MCQs exist in this chunk, respond with: []\n` +
          `- DO NOT STOP EARLY. Continue until you have extracted EVERY question from the text.\n\n` +
          `TEXT (chunk ${ci + 1}/${chunks.length}):\n${chunk}`;
        let attempts = 0;
        let success = false;
        while (attempts < 5 && !success) {
          try {
            const res = await capFetch(
              `https://generativelanguage.googleapis.com/v1beta/${model}:generateContent?key=${key}`,
              {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                  contents: [{ parts: [{ text: prompt }] }],
                  generationConfig: {
                    responseMimeType: 'application/json',
                    temperature: 0.1,
                    maxOutputTokens: 65536, // explicit max — prevents silent cutoff on large papers
                  },
                }),
              }
            );
            const data = await res.json();
            if (isQuotaError(data)) {
              attempts++;
              const waitMs = 15000 * attempts;
              addLog(`[${label}] chunk ${ci + 1} rate-limited (attempt ${attempts}/5). Waiting ${waitMs / 1000}s...`, 'warn');
              await new Promise(r => setTimeout(r, waitMs));
              continue;
            }
            // Fix (img4-8): deprecated model → skip to next model, not fatal
            if (data.error?.message?.includes('not found') || data.error?.message?.includes('not supported')) {
              addLog(`[${label}] ${shortName} not available — trying next model`, 'warn'); modelOk = false; break;
            }
            if (data.error) { addLog(`[${label}] ${shortName} error: ${data.error.message}`, 'warn'); modelOk = false; break; }
            const rawText: string = data.candidates?.[0]?.content?.parts?.[0]?.text || '';
            if (!rawText) { addLog(`[${label}] chunk ${ci + 1} empty response`, 'warn'); success = true; break; }
            const chunkQs = validateQuestions(parseQuestionsFromLLM(rawText));
            addLog(`[${label}] chunk ${ci + 1}/${chunks.length} ✓ ${chunkQs.length} questions`, 'ok');
            allQuestions.push(...chunkQs);
            success = true;
          } catch (e: any) {
            if (e.message?.toLowerCase().includes('rate') || e.message?.toLowerCase().includes('too many') || e.message?.includes('429')) {
              attempts++;
              const waitMs = 15000 * attempts;
              addLog(`[${label}] chunk ${ci + 1} rate-limited (attempt ${attempts}/5). Waiting ${waitMs / 1000}s...`, 'warn');
              await new Promise(r => setTimeout(r, waitMs));
              continue;
            }
            addLog(`[${label}] chunk ${ci + 1} error: ${e.message}`, 'warn');
            break;
          }
        }
        if (!success && modelOk) {
          // If we exhausted retries or failed, mark model as not OK for the remaining chunks.
          // Note: if modelOk was already set to false inside the loop, we just skip.
          if (attempts >= 5) {
             addLog(`[${label}] ${shortName} quota exhausted after retries — trying next`, 'warn');
             modelOk = false;
          }
        }
        if (!modelOk) break;
      }
      if (!modelOk) continue;
      if (allQuestions.length > 0) {
        addLog(`[${label}] ✓ ${allQuestions.length} questions extracted`, 'ok');
        return allQuestions;
      }
      addLog(`[${label}] ${shortName} returned no questions`, 'warn');
    }
    return null;
  };

  // ── GEMINI vision extraction (fallback — full PDF base64, handles scanned PDFs)
  const tryGemini = async (file: File, key: string): Promise<Question[] | null> => {
    addLog('Scanning available Gemini models...', 'info');
    const mRes = await capFetch(`https://generativelanguage.googleapis.com/v1beta/models?key=${key}`, { method: 'GET' });
    const mData = await mRes.json();
    if (mData.error) throw new Error(mData.error.message);

    const all: string[] = mData.models
      .filter((m: any) => m.supportedGenerationMethods?.includes('generateContent'))
      .map((m: any) => m.name as string);

    // Only gemini-2.5-flash — removed 2.0-flash and 2.0-flash-lite per user request.
    const priority = ['gemini-2.5-flash'];
    const sorted = [
      ...priority.map(p => all.find(n => n.includes(p))).filter(Boolean) as string[],
      ...all.filter(n => !priority.some(p => n.includes(p))),
    ];

    if (sorted.length === 0) throw new Error('No compatible Gemini model found');
    addLog(`Found ${sorted.length} Gemini model(s)`, 'info');

    const base64 = await new Promise<string>((resolve, reject) => {
      const r = new FileReader();
      r.readAsDataURL(file);
      r.onload = () => resolve((r.result as string).split(',')[1]);
      r.onerror = reject;
    });

    const prompt =
      `You are an MCQ extractor for Indian competitive exams (UPSC, SSC, GATE, NEET, JEE, State PSCs).\n` +
      `Read this PDF and extract ALL multiple-choice questions, including Hindi/bilingual text.\n` +
      `CRITICAL — DO NOT SKIP ANY QUESTION. Extract EVERY SINGLE MCQ from EVERY page.\n` +
      `IMPORTANT: Do NOT output just a subset of questions. Extract EVERY SINGLE QUESTION you find. If the PDF has 100 questions, your JSON array MUST contain 100 objects.\n` +
      `IMPORTANT RULES:\n` +
      `- Include ALL question types: standard MCQ, True/False, Assertion-Reason, Match-the-Column, Statement-based.\n` +
      `- If options are labelled 1/2/3/4 or a/b/c/d, normalize to A/B/C/D. Use roman numerals only if ALL options follow that scheme.\n` +
      `- For "Which of the following statements is/are correct?" questions, numbered sub-statements (1. 2. 3.) are PART of the question text, NOT option labels.\n` +
      `- The "answer" field MUST be one of the option letters (A, B, C, D, E, or F).\n` +
      `- For questions with diagrams/figures/maps/graphs: describe the visual briefly in the question text and prefix with "[Visual question]".\n` +
      `- For Assertion-Reason: preserve combo option text exactly (e.g. "Both A and R are true and R is the correct explanation of A").\n` +
      `- For Match-the-Column: encode List-I vs List-II in the question text. A/B/C/D options = matching combos.\n` +
      `- Preserve Hindi/Devanagari text exactly as it appears.\n` +
      `- Respond with ONLY a JSON array — no markdown, no explanation, no code fences.\n` +
      `- Each item: {"id":1,"question":"full question","options":{"A":"...","B":"...","C":"...","D":"..."},"answer":"B"}\n` +
      `- DO NOT STOP EARLY. Continue until you have extracted EVERY question from EVERY page of the PDF.\n` +
      `- Make sure every question has all options and the correct answer letter.`;

    for (const model of sorted) {
      const shortName = model.split('/').pop() || model;
      addLog(`Trying Gemini ${shortName}...`, 'info');
      const res = await capFetch(
        `https://generativelanguage.googleapis.com/v1beta/${model}:generateContent?key=${key}`,
        {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            contents: [{ parts: [{ text: prompt }, { inline_data: { mime_type: 'application/pdf', data: base64 } }] }],
            generationConfig: {
              responseMimeType: 'application/json',
              temperature: 0.1,
              maxOutputTokens: 65536, // prevent silent cutoff on large 100+ question papers
            },
          }),
        }
      );
      const data = await res.json();
      if (isQuotaError(data)) { addLog(`Gemini ${shortName} quota exceeded — trying next`, 'warn'); continue; }
      // Fix (img4-8): "not found for API version" means model is deprecated — skip, not fatal
      if (data.error?.message?.includes('not found') || data.error?.message?.includes('not supported')) {
        addLog(`Gemini ${shortName} not available — trying next`, 'warn'); continue;
      }
      if (data.error) throw new Error(data.error.message);

      const rawText: string = data.candidates[0].content.parts[0].text;
      const parsed = parseQuestionsFromLLM(rawText);
      const valid = validateQuestions(parsed);
      if (valid.length === 0) { addLog(`Gemini ${shortName} returned no valid questions`, 'warn'); continue; }

      addLog(`✓ Gemini ${shortName} extracted ${valid.length} questions`, 'ok');
      return valid;
    }
    return null;
  };

  // ── Deduplicate merged questions from parallel workers + overlap zones.
  // Uses first 120 chars of normalized question text (punctuation-stripped).
  // F6: Also detects bilingual duplicates by comparing option value fingerprints.
  // Guard: filter questions that look like answer-key entries (very short question, only 1-2 option words).
  const deduplicateQuestions = (questions: Question[]): Question[] => {
    const seen = new Set<string>();
    const optsSeen = new Map<string, number>();
    const result: Question[] = [];

    const normalize = (s: string) => s
      .replace(/^\[Visual question\]\s*/i, '')
      .replace(/^\[\.\.\.continued[^\]]*\]\s*/i, '')
      .toLowerCase()
      .replace(/[^\w\u0900-\u097F\s]/g, '') // strip punctuation but keep Devanagari
      .replace(/\s+/g, ' ')
      .trim()
      .slice(0, 120);

    const optFingerprint = (q: Question): string =>
      Object.values(q.options).map(v => v.trim().toLowerCase().replace(/\s+/g, '').slice(0, 30)).sort().join('|');

    for (const q of questions) {
      // Guard: reject questions that are actually answer-key rows (e.g. {"question":"1. C 2. B 3. A"})
      if (/^[\d\s.)\-A-Da-d,;|]+$/.test(q.question.trim()) && q.question.trim().length < 30) continue;
      // Guard: reject if options are all single letters or numbers (answer table artifact)
      const optVals = Object.values(q.options);
      if (optVals.every(v => /^[A-Da-d\d]$/.test(v.trim()))) continue;

      const normalized = normalize(q.question);
      if (normalized.length < 5) continue; // empty / garbage
      if (seen.has(normalized)) continue;
      seen.add(normalized);

      const optFp = optFingerprint(q);
      if (optFp.length > 5 && optsSeen.has(optFp)) {
        const existIdx = optsSeen.get(optFp)!;
        const existing = result[existIdx];
        // Prefer: non-truncated AR > longer question text
        const qTrunc = !!(q as any)._truncatedAR;
        const exTrunc = !!(existing as any)._truncatedAR;
        if (exTrunc && !qTrunc) { result[existIdx] = q; optsSeen.set(optFp, existIdx); }
        else if (!exTrunc && !qTrunc && q.question.length > existing.question.length) { result[existIdx] = q; }
        continue;
      }
      optsSeen.set(optFp, result.length);
      result.push(q);
    }
    return result;
  };

  // ── Main extraction: v11 — dynamic N-slice full-coverage parallel dispatch
  //
  // N = 2 × smallestPrimeFactor(totalPages), clamped to [5 … totalPages].
  // All N slices cover 100% of pages with no gaps and no overlap.
  // 5 AI workers (Groq×3 + Gemini-A + Gemini-B) each receive slices round-robin.
  // Every worker runs its assigned slices sequentially; all 5 workers fire in parallel.
  // Results merged + deduplicated. Falls back to Gemini vision if text layer is absent.
  //
  // ── smallestPrimeFactor: returns the 2nd-lowest factor of n (lowest factor > 1).
  const smallestPrimeFactor = (n: number): number => {
    if (n < 2) return 1;
    if (n % 2 === 0) return 2;
    for (let i = 3; i * i <= n; i += 2) if (n % i === 0) return i;
    return n; // n itself is prime
  };

  const startExtraction = async () => {
    if (!file) { alert('Please select a PDF first!'); return; }

    const gKey = apiKey.trim();
    const grKey = groqKey.trim();
    const nvKey = nvidiaKey.trim();

    if (!gKey && !grKey && !nvKey) { alert('Please enter at least one API key (Gemini, Groq, or NVIDIA)!'); return; }

    if (gKey) await dbSetSetting('gemini_api_key', gKey);
    if (grKey) await dbSetSetting('groq_api_key', grKey);
    if (nvKey) await dbSetSetting('nvidia_api_key', nvKey);

    setLoading(true);
    setLogs([{ msg: 'System initialized', kind: 'info' }]);

    try {
      const bothProviders = !!(gKey && grKey) || !!(grKey && nvKey) || !!(gKey && nvKey);

      // ── FULL-COVERAGE PARALLEL WORKER-POOL (any 2+ keys present)
      if (bothProviders) {
        addLog('Extracting PDF pages...', 'info');
        const { pages, scannedPageCount, scannedPageImages } = await extractPdfPages(file);
        const totalPages = pages.length;

        // OCR scanned pages — prefer NVIDIA vision, fall back to Gemini
        let patchedPages = [...pages];
        if (scannedPageCount > 0 && Object.keys(scannedPageImages).length > 0) {
          const pct = Math.round((scannedPageCount / totalPages) * 100);
          addLog(`⚠ ${scannedPageCount}/${totalPages} pages image-only/garbled (${pct}%). Running OCR...`, 'warn');
          const ocrResults = nvKey
            ? await ocrScannedPagesWithNvidia(scannedPageImages, nvKey, addLog)
            : await ocrScannedPagesWithGemini(scannedPageImages, gKey, addLog);
          for (const [idxStr, text] of Object.entries(ocrResults)) {
            patchedPages[Number(idxStr)] = text;
          }
          addLog(`[OCR] Patched ${Object.keys(ocrResults).length} pages back into text layer`, 'info');
        } else if (scannedPageCount > 0) {
          const pct = Math.round((scannedPageCount / totalPages) * 100);
          addLog(
            `⚠ ${scannedPageCount}/${totalPages} pages appear image-only (${pct}%). ` +
            `${pct > 50 ? 'Switching to Gemini vision...' : 'Text workers will skip blank pages.'}`,
            'warn'
          );
        }

        // F3: Strip repeated headers, footers, watermarks
        patchedPages = stripHeadersFooters(patchedPages);

        // F5: Pre-extract answer key from last pages
        const akResult = extractAnswerKey(patchedPages);
        const answerKey = akResult?.answers ?? null;
        const akTailPages = akResult?.tailPageCount ?? 0;
        if (answerKey) {
          addLog(`📋 Found answer key (${Object.keys(answerKey).length} answers) spanning last ${akTailPages} page(s) — excluding from AI slices`, 'ok');
        }

        // Exclude answer-key pages from what we send to AI so models don't extract
        // answer-key rows as fake questions
        const slicePages = akTailPages > 0 ? patchedPages.slice(0, -akTailPages) : patchedPages;
        const fullText = slicePages.join('\n').trim();
        const pdfOk = fullText.length >= 100;
        if (pdfOk) {
          const allSlices = buildSlicesWithOverlap(slicePages, 12000);

          addLog(
            `⚡ v14 (failover): ${totalPages} pages → ${allSlices.length} slices ` +
            `[${grKey ? 'Groq×3' : ''}${gKey ? ' + Gemini' : ''}${nvKey ? ' + NVIDIA×2' : ''}] with cross-AI failover`,
            'info'
          );

          // ── Centralized failover dispatch (v14)
          // Each slice is tried by its primary AI. If it returns 0 questions (failed /
          // rate-limited / truncated), the slice is retried by the next available AI.
          // Order: Groq(round-robin 3 models) → Gemini-text → NVIDIA(2 models) → repeat.
          type SliceCaller = (sliceText: string, sliceIdx: number) => Promise<Question[]>;
          const aiCallers: { name: string; call: SliceCaller }[] = [];

          if (grKey) {
            GROQ_MODELS.forEach(({ model, maxTokens }) =>
              aiCallers.push({
                name: `Groq:${model.split('/').pop()!.split('-').slice(0, 3).join('-')}`,
                call: (text, idx) => groqCallOne(text, idx, allSlices.length, model, maxTokens, grKey).catch(() => []),
              })
            );
          }
          if (gKey) {
            aiCallers.push({
              name: 'Gemini-text',
              call: async (text, idx) => {
                const r = await tryGeminiText(text, gKey, `Failover-G-s${idx + 1}`);
                return r ?? [];
              },
            });
          }
          if (nvKey) {
            NVIDIA_MODELS.forEach(({ model, maxTokens }) =>
              aiCallers.push({
                name: `NVIDIA:${model.split('/').pop()!}`,
                call: (text, idx) => nvidiaCallOne(text, idx, allSlices.length, model, maxTokens, nvKey).catch(() => []),
              })
            );
          }

          // Run all slices in parallel; each uses its own AI with failover to next
          const sliceResults: Question[][] = new Array(allSlices.length).fill([]);
          await Promise.all(
            allSlices.map(async (sliceText, idx) => {
              if (aiCallers.length === 0) return;
              const primaryIdx = idx % aiCallers.length;
              let qs: Question[] = [];

              qs = await aiCallers[primaryIdx].call(sliceText, idx);

              if (qs.length === 0) {
                for (let attempt = 1; attempt < aiCallers.length; attempt++) {
                  const fallbackIdx = (primaryIdx + attempt) % aiCallers.length;
                  const fallback = aiCallers[fallbackIdx];
                  addLog(`[Failover] Slice ${idx + 1} → ${fallback.name} (primary returned 0)`, 'warn');
                  qs = await fallback.call(sliceText, idx);
                  if (qs.length > 0) {
                    addLog(`[Failover] Slice ${idx + 1} recovered: ${qs.length} Qs via ${fallback.name}`, 'ok');
                    break;
                  }
                }
                if (qs.length === 0)
                  addLog(`[Failover] Slice ${idx + 1} — all AIs returned 0 Qs (may be answer-key-only or blank)`, 'warn');
              }

              sliceResults[idx] = qs;
            })
          );

          const allRaw = sliceResults.flat();
          const merged = deduplicateQuestions(allRaw);
          addLog(`✓ All slices done — raw: ${allRaw.length} → deduped: ${merged.length}`, merged.length > 0 ? 'ok' : 'warn');

          if (merged.length > 0) {
            // Sort by original PDF question id before backfill
            merged.sort((a, b) => a.id - b.id);

            if (answerKey) {
              let backfilled = 0;
              let positional = 0;
              merged.forEach((q, posIdx) => {
                if (q.answer && q.answer !== '') return; // already has answer
                // 1st: match by original PDF question number
                const keyAns = answerKey[q.id];
                if (keyAns && q.options[keyAns]) { q.answer = keyAns; backfilled++; return; }
                // 2nd: positional fallback — answer key entry at position (posIdx+1)
                // Handles cases where model renumbered questions starting from 1
                const posAns = answerKey[posIdx + 1];
                if (posAns && q.options[posAns]) { q.answer = posAns; positional++; }
              });
              if (backfilled > 0) addLog(`📋 Backfilled ${backfilled} answers by question ID`, 'ok');
              if (positional > 0) addLog(`📋 Positionally backfilled ${positional} answers (ID mismatch fallback)`, 'warn');

              const keyCount = Object.keys(answerKey).length;
              if (keyCount > merged.length * 1.2) {
                addLog(`⚠ Answer key has ${keyCount} entries but only ${merged.length} Qs extracted — some questions may be missing`, 'warn');
              }
            }

            // Log which questions still have no answer so user knows what was dropped
            const noAnswer = merged.filter(q => !q.answer || q.answer === '');
            if (noAnswer.length > 0) {
              addLog(`⚠ ${noAnswer.length} question(s) have no answer after backfill — dropping: ${noAnswer.map(q => `Q${q.id}`).slice(0, 10).join(', ')}${noAnswer.length > 10 ? '…' : ''}`, 'warn');
            }

            const fullyValid = merged.filter(q => q.answer && q.answer !== '');
            // Renumber sequentially 1…N so IDs are always contiguous in the test UI
            fullyValid.forEach((q, i) => { q.id = i + 1; });

            if (fullyValid.length > 0) {
              await finishExtraction(fullyValid, 'groq+gemini', file.name);
              return;
            }
          }

          // Fallback: scanned PDF — try Gemini vision
          addLog('Text extraction yielded nothing — trying Gemini vision (scanned PDF?)...', 'warn');
          const visionQs = await tryGemini(file, gKey);
          if (visionQs && visionQs.length > 0) {
            await finishExtraction(visionQs, 'gemini-vision', file.name);
            return;
          }
          throw new Error('All providers failed. PDF may be image-only with no text layer.');

        } else {
          addLog(`PDF appears scanned (${scannedPageCount}/${patchedPages.length} blank pages) — using Gemini vision...`, 'warn');
          const valid = await tryGemini(file, gKey);
          if (valid && valid.length > 0) { await finishExtraction(valid, 'gemini-vision', file.name); return; }
          throw new Error('Gemini vision extracted no valid questions from this PDF.');
        }
      }

      // ── SINGLE PROVIDER paths
      if (grKey) {
        addLog('Extracting PDF text for Groq...', 'info');
        const { pages, scannedPageCount, scannedPageImages } = await extractPdfPages(file);
        let patchedPages = [...pages];
        if (scannedPageCount > 0 && Object.keys(scannedPageImages).length > 0)
          addLog(`⚠ ${scannedPageCount}/${pages.length} pages appear image-only — add a Gemini key for OCR/vision fallback.`, 'warn');
        patchedPages = stripHeadersFooters(patchedPages);
        const akResult = extractAnswerKey(patchedPages);
        const answerKey = akResult?.answers ?? null;
        const akTailPages = akResult?.tailPageCount ?? 0;
        if (answerKey) addLog(`📋 Found answer key (${Object.keys(answerKey).length} answers) — excluding last ${akTailPages} page(s) from slices`, 'ok');
        const slicePages = akTailPages > 0 ? patchedPages.slice(0, -akTailPages) : patchedPages;
        const fullText = slicePages.join('\n').trim();
        if (fullText.length >= 100) {
          const slices = buildSlicesWithOverlap(slicePages, 12000);
          addLog(`PDF extracted — ${slices.length} slices (overlap) → Groq...`, 'info');
          let qs = await tryGroq(slices, grKey);
          if (answerKey) {
            qs.sort((a, b) => a.id - b.id);
            let bf = 0, pf = 0;
            qs.forEach((q, i) => {
              if (q.answer && q.answer !== '') return;
              const ka = answerKey[q.id]; if (ka && q.options[ka]) { q.answer = ka; bf++; return; }
              const pa = answerKey[i + 1]; if (pa && q.options[pa]) { q.answer = pa; pf++; }
            });
            if (bf > 0) addLog(`📋 Backfilled ${bf} answers by question ID`, 'ok');
            if (pf > 0) addLog(`📋 Positionally backfilled ${pf} answers`, 'warn');
          }
          const noAnswer = qs.filter(q => !q.answer || q.answer === '');
          if (noAnswer.length > 0) addLog(`⚠ Dropping ${noAnswer.length} question(s) with no answer: ${noAnswer.map(q => `Q${q.id}`).slice(0,10).join(', ')}`, 'warn');
          qs = qs.filter(q => q.answer && q.answer !== '');
          qs.forEach((q, i) => { q.id = i + 1; });
          if (qs.length > 0) { await finishExtraction(qs, 'groq', file.name); return; }
          throw new Error('Groq could not extract questions. Try adding a Gemini key for vision fallback.');
        }
        throw new Error('PDF text extraction failed — file may be scanned. Add a Gemini key to use vision mode.');
      }

      if (nvKey) {
        addLog('Extracting PDF text for NVIDIA...', 'info');
        const { pages } = await extractPdfPages(file);
        let patchedPages = stripHeadersFooters([...pages]);
        const akResult = extractAnswerKey(patchedPages);
        const answerKey = akResult?.answers ?? null;
        const akTailPages = akResult?.tailPageCount ?? 0;
        if (answerKey) addLog(`📋 Found answer key (${Object.keys(answerKey).length} answers) — excluding last ${akTailPages} page(s)`, 'ok');
        const slicePages = akTailPages > 0 ? patchedPages.slice(0, -akTailPages) : patchedPages;
        const fullText = slicePages.join('\n').trim();
        if (fullText.length >= 100) {
          const slices = buildSlicesWithOverlap(slicePages, 12000);
          addLog(`PDF extracted — ${slices.length} slices → NVIDIA...`, 'info');
          const allQs: Question[] = [];
          for (let i = 0; i < slices.length; i++) {
            const { model, maxTokens } = NVIDIA_MODELS[i % NVIDIA_MODELS.length];
            try {
              const qs = await nvidiaCallOne(slices[i], i, slices.length, model, maxTokens, nvKey);
              allQs.push(...qs);
            } catch {}
          }
          let deduped = deduplicateQuestions(allQs);
          deduped.sort((a, b) => a.id - b.id);
          if (answerKey) {
            let bf = 0, pf = 0;
            deduped.forEach((q, i) => {
              if (q.answer && q.answer !== '') return;
              const ka = answerKey[q.id]; if (ka && q.options[ka]) { q.answer = ka; bf++; return; }
              const pa = answerKey[i + 1]; if (pa && q.options[pa]) { q.answer = pa; pf++; }
            });
            if (bf > 0) addLog(`📋 Backfilled ${bf} answers by question ID`, 'ok');
            if (pf > 0) addLog(`📋 Positionally backfilled ${pf} answers`, 'warn');
          }
          const noAnswer = deduped.filter(q => !q.answer || q.answer === '');
          if (noAnswer.length > 0) addLog(`⚠ Dropping ${noAnswer.length} question(s) with no answer: ${noAnswer.map(q=>`Q${q.id}`).slice(0,10).join(', ')}`, 'warn');
          deduped = deduped.filter(q => q.answer && q.answer !== '');
          deduped.forEach((q, i) => { q.id = i + 1; });
          if (deduped.length > 0) { await finishExtraction(deduped, 'nvidia', file.name); return; }
          throw new Error('NVIDIA could not extract questions. Try adding a Gemini or Groq key.');
        }
        throw new Error('PDF text extraction failed. Add a Gemini key for vision mode.');
      }

      if (gKey) {
        addLog('Using Gemini (vision-based PDF reading)...', 'info');
        const valid = await tryGemini(file, gKey);
        if (valid && valid.length > 0) { await finishExtraction(valid, 'gemini', file.name); return; }
        throw new Error('Gemini extracted no valid questions from this PDF.');
      }

      throw new Error('All extraction methods failed. Check your API keys and try again.');

    } catch (err: any) {
      addLog(`Error: ${err.message || String(err)}`, 'err');
      alert(`Extraction failed:\n${err.message || String(err)}`);
      setLoading(false);
    }

Question: ${q.question}
Options: ${Object.entries(q.options).map(([k, v]) => `${k}. ${v}`).join(' | ')}
Correct: ${q.answer}. ${q.options[q.answer]}`;

                    const scoreColor = best === null ? 'var(--muted)' : best >= 70 ? 'var(--green)' : best >= 40 ? 'var(--yellow)' : 'var(--red)';
                    return (
                      <div key={t.id} className="saved-item" style={{ animationDelay: `${i * 0.05}s` }}>
                        <div className="saved-icon"><FileText size={16} color="var(--cyan)" /></div>
                        <div className="saved-info">
                          <div className="saved-name">{t.name.replace('.pdf', '')}</div>
                          <div className="saved-meta">
                            <span><CheckCircle size={10} color="var(--green)" />{t.questions.length} Qs</span>
                            {best !== null ? <span style={{ color: scoreColor, fontWeight: 600 }}>Best: {best}%</span> : <span>Not attempted</span>}
                            {attempts.length > 0 && <span>{attempts.length}× tried</span>}
                          </div>
                        </div>
                        <div className="saved-actions">
                          <button className="btn-load" onClick={() => loadSavedTest(t)} title="Start Test">
                            <PlayCircle size={11} style={{ display: 'inline', marginRight: 4, verticalAlign: 'middle' }} />Start
                          </button>
                          <button className="btn-load" style={{ background: 'rgba(0,229,255,0.05)', borderColor: 'rgba(0,229,255,0.15)' }}
                            onClick={() => setAnalyticsTestId(t.id)} title="Analytics">📊</button>
                          <button className="btn-del" onClick={() => handleDeleteTest(t.id)} title="Delete"><Trash2 size={13} /></button>
                        </div>
                      </div>
                    );
                  })}
                </div>
              )}
            </>
          )}
        </div>
      )}
    </div>
  );

  // ── Test screen
  const renderTest = () => {
    if (!questions.length) return null;
    const q = questions[qIndex];
    const answered = Object.keys(selections).length;
    const flaggedCount = Object.values(flagged).filter(Boolean).length;
    const visitedPct = ((qIndex + 1) / questions.length) * 100;
    const answeredPct = (answered / questions.length) * 100;
    const isUrgent = clock < 300;
    const isCorrectNow = studyMode && selections[q.id] === q.answer;
    const isWrongNow   = studyMode && !!selections[q.id] && selections[q.id] !== q.answer;
    return (
      <div className="test-layout">
        <div className="sidebar">
          <div className="s-card">
            <div className="timer-lbl"><Clock size={10} />Time Left</div>
            <div className={`timer-val${isUrgent ? ' urgent' : ''}`}>{formatTime(clock)}</div>
            {discoveredModel && <div className="model-badge"><Zap size={9} />{discoveredModel}</div>}
            {studyMode && <div style={{ marginTop: 8, fontSize: 10, color: 'var(--yellow)', fontFamily: 'JetBrains Mono', padding: '3px 8px', background: 'rgba(255,230,0,0.08)', borderRadius: 6, border: '1px solid rgba(255,230,0,0.2)', display: 'inline-flex', alignItems: 'center', gap: 4 }}>📖 Study Mode</div>}
            {activeTestName && <div style={{ marginTop: 8, fontSize: 10, color: 'var(--muted)', fontFamily: 'Rajdhani', lineHeight: 1.4, wordBreak: 'break-word' }}>📄 {activeTestName.replace('.pdf', '')}</div>}
          </div>
          <div className="s-card">
            <div className="prog-lbl">Progress</div>
            <div className="prog-row">
              <div className="prog-top"><span>Visited</span><strong>{qIndex + 1} / {questions.length}</strong></div>
              <div className="prog-track"><div className="prog-fill v" style={{ width: `${visitedPct}%` }} /></div>
            </div>
            <div className="prog-row">
              <div className="prog-top"><span>Answered</span><strong>{answered} / {questions.length}</strong></div>
              <div className="prog-track"><div className="prog-fill a" style={{ width: `${answeredPct}%` }} /></div>
            </div>
            <div style={{ display: 'flex', justifyContent: 'space-between', fontSize: 10.5, color: 'var(--muted)', marginTop: 5 }}>
              <span>{questions.length - answered} remaining</span>
              {flaggedCount > 0 && <span style={{ color: 'var(--yellow)' }}>🚩 {flaggedCount} flagged</span>}
            </div>
          </div>

          {/* ── Question Navigator Grid */}
          <div className="s-card">
            <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'space-between', marginBottom: 8 }}>
              <div className="prog-lbl" style={{ margin: 0 }}>Navigator</div>
              <button onClick={() => setShowNavigator(s => !s)}
                style={{ background: 'none', border: 'none', color: 'var(--muted)', fontSize: 9, fontFamily: 'Orbitron', cursor: 'pointer', letterSpacing: 1 }}>
                {showNavigator ? '▲ Hide' : '▼ Show'}
              </button>
            </div>
            {showNavigator && (
              <>
                <div className="nav-legend">
                  <div className="leg-item"><div className="leg-dot" style={{ background: 'var(--surface)', border: '1px solid var(--border2)' }} />Unseen</div>
                  <div className="leg-item"><div className="leg-dot" style={{ background: 'var(--green-dim)', border: '1px solid rgba(0,255,136,0.25)' }} />Done</div>
                  <div className="leg-item"><div className="leg-dot" style={{ background: 'var(--cyan-dim)', border: '1px solid rgba(0,229,255,0.4)' }} />Current</div>
                  <div className="leg-item"><div className="leg-dot" style={{ background: 'transparent', border: '1px solid var(--yellow)' }} />Flagged</div>
                </div>
                <div className="q-grid">
                  {questions.map((qs, i) => {
                    const isAns = !!selections[qs.id];
                    const isFlg = !!flagged[qs.id];
                    const isCur = i === qIndex;
                    return (
                      <button key={qs.id}
                        className={`q-dot${isCur ? ' cur' : isAns ? ' ans' : ''}${isFlg ? ' flg' : ''}`}
                        style={isFlg ? { outline: '1.5px solid var(--yellow)' } : {}}
                        onClick={() => setQIndex(i)}
                        title={`Q${qs.id}${isFlg ? ' (Flagged)' : ''}`}>
                        {qs.id}
                      </button>
                    );
                  })}
                </div>
              </>
            )}
            {!showNavigator && (
              <div style={{ fontSize: 10, color: 'var(--muted)' }}>
                {answered}/{questions.length} answered · tap to expand
              </div>
            )}
          </div>

          <button className="end-btn" onClick={endExam}><XCircle size={14} />End Test</button>
        </div>
        <div className="q-card" key={animKey} style={{ animation: 'scaleIn 0.28s cubic-bezier(.16,1,.3,1)' }}>
          <div className="q-head">
            <span className="q-badge">Q {q.id} / {questions.length}</span>
            <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
              <button
                className={`flag-btn${flagged[q.id] ? ' flagged' : ''}`}
                onClick={() => setFlagged(prev => ({ ...prev, [q.id]: !prev[q.id] }))}
                title="Flag for Review (F)">
                {flagged[q.id] ? '🚩 Flagged' : '⚑ Flag'}
              </button>
              <span className="q-status" style={{ color: selections[q.id] ? 'var(--green)' : 'var(--muted)' }}>
                {selections[q.id] ? <><CheckCircle size={13} />Answered</> : 'Unanswered'}
              </span>
            </div>
          </div>
          <p className="q-text">{q.question}</p>
          {(q as any)._hasVisual && (
            <div style={{
              background: 'rgba(255,230,0,0.08)', border: '1px solid rgba(255,230,0,0.25)',
              borderRadius: 8, padding: '7px 12px', marginBottom: 10,
              fontSize: 11.5, color: 'var(--yellow)', display: 'flex', alignItems: 'center', gap: 6
            }}>
              <AlertTriangle size={13} color="var(--yellow)" />
              This question may reference a diagram or figure from the original PDF.
            </div>
          )}
          <div className="opts">
            {Object.entries(q.options).map(([k, v], idx) => {
              const isSel = selections[q.id] === k;
              const showResult = studyMode && !!selections[q.id];
              const isCorrectOpt = k === q.answer;
              let optStyle: React.CSSProperties = {};
              if (showResult) {
                if (isCorrectOpt) optStyle = { borderColor: 'var(--green)', background: 'rgba(0,255,136,0.08)' };
                else if (isSel && !isCorrectOpt) optStyle = { borderColor: 'var(--red)', background: 'rgba(255,51,85,0.07)' };
              }
              return (
                <button key={k} className={`opt-btn${isSel ? ' sel' : ''}`}
                  style={{ animationDelay: `${idx * 0.055}s`, ...optStyle }}
                  onClick={() => setSelections(prev => ({ ...prev, [q.id]: k }))}>
                  <span className="opt-key">{k}</span>
                  <span className="opt-text">{v}</span>
                  {showResult && isCorrectOpt && <CheckCircle size={14} color="var(--green)" style={{ marginLeft: 'auto', flexShrink: 0 }} />}
                  {showResult && isSel && !isCorrectOpt && <XCircle size={14} color="var(--red)" style={{ marginLeft: 'auto', flexShrink: 0 }} />}
                </button>
              );
            })}
          </div>

          {/* Study Mode instant feedback */}
          {studyMode && selections[q.id] && (
            <div className={`study-reveal ${isCorrectNow ? 'study-correct' : 'study-wrong'}`}>
              {isCorrectNow
                ? `✓ Correct! ${q.answer}. ${q.options[q.answer]}`
                : `✗ Wrong — Correct answer: ${q.answer}. ${q.options[q.answer]}`}
            </div>
          )}

          <div className="q-nav">
            <button className="nav-btn" disabled={qIndex === 0} onClick={() => setQIndex(Math.max(0, qIndex - 1))}><ChevronLeft size={15} />Prev</button>
            <button className="nav-btn nxt" onClick={() => qIndex === questions.length - 1 ? endExam() : setQIndex(qIndex + 1)}>
              {qIndex === questions.length - 1 ? 'Finish ✓' : <>Next<ChevronRight size={15} /></>}
            </button>
          </div>

          {/* Keyboard hint */}
          <div className="kbd-hint">
            <div className="kbd-chip"><span className="kbd">A</span><span className="kbd">B</span><span className="kbd">C</span><span className="kbd">D</span> answer</div>
            <div className="kbd-chip"><span className="kbd">←</span><span className="kbd">→</span> navigate</div>
            <div className="kbd-chip"><span className="kbd">F</span> flag</div>
          </div>
        </div>
      </div>
    );
  };

  // ── Results screen
  const renderResults = () => {
    let score = 0, skipped = 0;
    questions.forEach(q => { if (!selections[q.id]) skipped++; else if (selections[q.id] === q.answer) score++; });
    const wrong = questions.length - score - skipped;
    const pct = questions.length > 0 ? Math.round((score / questions.length) * 100) : 0;
    const timeTaken = startTime > 0 ? Math.round((Date.now() - startTime) / 1000) : 0;
    const wrongQuestions = questions.filter(q => selections[q.id] && selections[q.id] !== q.answer);

    if (activeTestId && questions.length > 0 && !resultsSaved.current) {
      resultsSaved.current = true;
      saveAttemptRecord(activeTestId, activeTestName, pct, score, wrong, skipped, questions.length, timeTaken);
    }

    return (
      <div className="results-wrap">
        <div className="results-bar no-print">
          <button className="btn-ghost" onClick={restartApp}><RotateCcw size={14} />New Exam</button>
          <div style={{ display: 'flex', gap: 8, flexWrap: 'wrap' }}>
            {wrongQuestions.length > 0 && (
              <button className="btn-retry" onClick={() => retryWrongQuestions(wrongQuestions)}>
                ↺ Retry {wrongQuestions.length} Wrong
              </button>
            )}
            {activeTestId && (
              <button className="btn-ghost" style={{ borderColor: 'rgba(0,229,255,0.2)', color: 'var(--cyan)' }}
                onClick={() => { setAnalyticsTestId(activeTestId); restartApp(true); }}>📊 Analytics</button>
            )}
            <button className="btn-save" onClick={saveResults}><Download size={14} />Save PDF</button>
          </div>
        </div>
        <div ref={resultsRef}>
          <div className="score-hero">
            <div className="score-pct">{pct}%</div>
            <p className="score-sub">{score} correct out of {questions.length} questions</p>
            <div className="stats-row">
              <div className="stat-chip c"><CheckCircle size={13} color="var(--green)" /><span className="v">{score}</span> Correct</div>
              <div className="stat-chip w"><XCircle size={13} color="var(--red)" /><span className="v">{wrong}</span> Wrong</div>
              <div className="stat-chip s"><span className="v">{skipped}</span> Skipped</div>
            </div>
          </div>
          <div className="review-list">
            {questions.map((q, i) => {
              const ok = selections[q.id] === q.answer;
              const sk = !selections[q.id];
              return (
                <div key={q.id} className={`review-item ${sk ? 'skipped' : ok ? 'correct' : 'wrong'}`}
                  style={{ animationDelay: `${Math.min(i * 0.035, 0.5)}s` }}>
                  <p className="review-q">Q{q.id}: {q.question}</p>
                  <div className="ans-grid">
                    <div className="ans-box">
                      <div className="ans-lbl">Your Answer</div>
                      <div className="ans-val" style={{ color: sk ? 'var(--muted)' : ok ? 'var(--green)' : 'var(--red)' }}>
                        {sk ? '— Skipped' : `${selections[q.id]}. ${q.options[selections[q.id]] || ''}`}
                      </div>
                    </div>
                    <div className="ans-box">
                      <div className="ans-lbl">Correct Answer</div>
                      <div className="ans-val" style={{ color: 'var(--green)' }}>{q.answer}. {q.options[q.answer]}</div>
                    </div>
                  </div>
                  {aiExplanations[q.id] ? (
                    <div className="ai-box">
                      <div className="ai-box-title"><Zap size={10} />AI Explanation</div>
                      {aiExplanations[q.id]}
                    </div>
                  ) : (
                    <button className="ai-btn no-print" disabled={!!loadingExplanation[q.id]} onClick={() => fetchExplanation(q)}>
                      <Zap size={12} />{loadingExplanation[q.id] ? 'Thinking…' : 'Explain with AI'}
                    </button>
                  )}
                </div>
              );
            })}
          </div>
        </div>
      </div>
    );
  };

  return (
    <div style={{ minHeight: '100vh', background: 'var(--bg)' }}>
      {screen === 'upload' && renderUpload()}
      {screen === 'test' && renderTest()}
      {screen === 'results' && renderResults()}
    </div>
  );
}
